\def\Syntax #1{%
	\setbox\gnBoxA=\hbox{$#1$}%
	\gnArgHgt=\ht\gnBoxA%
	\ifnum \gnArgHgt<\gnCornerHgt
	\gnArgHgt=0pt%
	\else
	\advance \gnArgHgt by -\gnCornerHgt%
	\fi
	\raise\gnArgHgt\hbox{$\ulcorner$} \box\gnBoxA %
	\raise\gnArgHgt\hbox{$\urcorner$}}
